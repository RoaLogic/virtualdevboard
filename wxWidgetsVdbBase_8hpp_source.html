186 virtual void SetAngle(double angle)
187 {
188 double radAngle;
189
190 while (angle >= 360) angle -= 360;
191 while (angle <= -360) angle += 360;
192 _angle = angle;
193
194 //XY: X increases left-to-right, Y increases bottom-to-top
195 //Video: X increases left-to-right, Y increases top-to-bottom
196 //Invert angle to ensure counter-clockwise rotation
197 angle = -_angle;
198
199 radAngle = AngleToRadians(angle);
200
201 _sinAngle = std::sin(radAngle);
202 _cosAngle = std::cos(radAngle);
203 }

209 double AngleToRadians(double angle) const { return (2.0 * std::numbers::pi * angle / 360.0); }

214 distanceSize Rotate(const distanceSize size) const
215 {
216 distanceSize newSize;
217 double absSin = (_sinAngle < 0) ? -_sinAngle : _sinAngle;
218 double absCos = (_cosAngle < 0) ? -_cosAngle : _cosAngle;
219
220 //transpose width and height
221 newSize.width = (absCos * size.width ) + (absSin * size.height);
222 newSize.height = (absCos * size.height) + (absSin * size.width );
223
224 return newSize;
225 }
226 wxSize wxRotate(const distanceSize sz) const
227 {
228 return wxDistanceSize(Rotate(sz),this);
229 }

235 distancePoint Rotate(const cDistance x, const cDistance y) const
236 {
237 //normalize around rotation point (center of logical canvas) before rotation
238 cDistance x_offset = (x - _origin.x) - GetDeviceSize().width/2.0;
239 cDistance y_offset = (y - _origin.y) - GetDeviceSize().height/2.0;
240
241 distancePoint newPoint;
242
243 //transpose x and y
244 //Canvas size changed due to rotation, move to center of resized canvas
245 newPoint.x = (x_offset * _cosAngle) - (y_offset * _sinAngle) + Rotate(GetDeviceSize()).width/2.0;
246 newPoint.y = (x_offset * _sinAngle) + (y_offset * _cosAngle) + Rotate(GetDeviceSize()).height/2.0;
247
248 return newPoint;
249 }
