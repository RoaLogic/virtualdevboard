146 token_t getToken()
147 {
148 for (;;)
149 {
150 int c = istream->get();
151
152 if (c == EOF)
153 return tokEOF;
154
155 switch (c)
156 {
157 //Skip white spaces
158 case '\n': lineno++;
159 case ' ':
160 case '\t':
161 case '\r':
162 continue;
163
164
165 //Handle multi-line comments
166 case '%':
167 do
168 {
169 c = istream->get();
170 }
171 while (c != EOF && c != '%');
172 continue;
173
174
175 case '-':
176 if (istream->peek() == '-')
177 {
178 //comment
179 do
180 {
181 c = istream->get();
182 }
183 while (c != EOF && c != '\n');
184 continue;
185 }
186 else
187 return tokMinus;
188
189
190 case '=':
191 return tokEquals;
192
193
194 case ':':
195 return tokColon;
196
197
198 case ';':
199 return tokSemicolon;
200
201
202 case '[':
203 return tokBracketLeft;
204
205
206 case ']':
207 return tokBracketRight;
208
209
210 case '.':
211 if (istream->peek() == '.')
212 {
213 istream->get();
214 return tokDotDot;
215 }
216 else
217 return tokDot;
218
219
220 //Alphanumeric tokens
221 //case 0...9 would be so nice here
222 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
223 case 'g': case 'h': case 'i': case 'j': case 'k': case 'l':
224 case 'm': case 'n': case 'o': case 'p': case 'q': case 'r':
225 case 's': case 't': case 'u': case 'v': case 'w': case 'x':
226 case 'y': case 'z':
227 case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
228 case 'G': case 'H': case 'I': case 'J': case 'K': case 'L':
229 case 'M': case 'N': case 'O': case 'P': case 'Q': case 'R':
230 case 'S': case 'T': case 'U': case 'V': case 'W': case 'X':
231 case 'Y': case 'Z':
232 case '0': case '1': case '2': case '3': case '4': case '5':
233 case '6': case '7': case '8': case '9':
234 case '_':
235 {
236 std::string sId;
237
238 for (;;)
239 {
240 //append character (as char) to sId
241 sId += (char)c;
242
243 //get next character
244 c = istream->get();
245
246 if (c == EOF)
247 break;
248
249
250 switch (c)
251 {
252 //keep adding alphanum characters
253 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
254 case 'g': case 'h': case 'i': case 'j': case 'k': case 'l':
255 case 'm': case 'n': case 'o': case 'p': case 'q': case 'r':
256 case 's': case 't': case 'u': case 'v': case 'w': case 'x':
257 case 'y': case 'z':
258 case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
259 case 'G': case 'H': case 'I': case 'J': case 'K': case 'L':
260 case 'M': case 'N': case 'O': case 'P': case 'Q': case 'R':
261 case 'S': case 'T': case 'U': case 'V': case 'W': case 'X':
262 case 'Y': case 'Z':
263 case '0': case '1': case '2': case '3': case '4': case '5':
264 case '6': case '7': case '8': case '9':
265 case '_':
266 continue;
267
268
269 //Not an alphanum, so not part of the string
270 //Put it back into the stream
271 default:
272 istream->unget();
273 break;
274 }
275
276 //exit append-loop
277 break;
278
279 } //end append-for-loop
280
281
282 //We got sId, now what is it?
283
284 //is it a number?
285 const char* c_str = sId.c_str();
286 char* endptr;
287 tokValue = strtol(c_str, &endptr, lexRadix);
288 if (endptr != c_str && *endptr == '\0')
289 return tokNumber;
290
291 //no, not a number
292 //is it a token?
293 if (sId == "ADDRESS_RADIX")
294 return tokAddressRadix;
295 if (sId == "BEGIN")
296 return tokBegin;
297 if (sId == "BIN")
298 return tokBin;
299 if (sId == "CONTENT")
300 return tokContent;
301 if (sId == "DATA_RADIX")
302 return tokDataRadix;
303 if (sId == "DEC")
304 return tokDec;
305 if (sId == "DEPTH")
306 return tokDepth;
307 if (sId == "END")
308 return tokEnd;
309 if (sId == "HEX")
310 return tokHex;
311 if (sId == "OCT")
312 return tokOct;
313 if (sId == "UNS")
314 return tokDec;
315 if (sId == "WIDTH")
316 return tokWidth;
317
318 return tokUnknown;
319 } //end alphanum
320
321
322 default:
323 return tokUnknown;
324
325 } //end switch
326 } //end for
327 }
328 };
