109 void toggle(void)
110 {
111 //toggle clock
112 _clk = !_clk;
113
114 //Update TimeKeep
115 _timeToNextEvent = _clk ? _highPeriod : _lowPeriod;
116
117 //call routines waiting for posedge/negedge
118 if (_clk)
119 {
120 resumeWaitForPosedge();
121 }
122 else
123 {
124 resumeWaitForNegedge();
125 }
126 }

143 void resumeWaitForPosedge()
144 {
145 if(!posedgeQueue.empty())
146 {
147 // Create a new temporary empty queue
148 std::queue<std::coroutine_handle<>> queueCopy;
149
150 // Swap empty queue and posedgeQueue
151 posedgeQueue.swap(queueCopy);
152
153 do
154 {
155 //pop the function from the queue
156 std::coroutine_handle<> h = queueCopy.front();
157 queueCopy.pop();
158
159 //resume the coroutine
160 h.resume();
161 } while (!queueCopy.empty());
162 }
163 }
164
180 void resumeWaitForNegedge()
181 {
182 if(!negedgeQueue.empty())
183 {
184 // Create a new temporary empty queue
185 std::queue<std::coroutine_handle<>> queueCopy;
186
187 // Swap empty queue and posedgeQueue
188 negedgeQueue.swap(queueCopy);
189
190 do
191 {
192 //pop the function from the queue
193 std::coroutine_handle<> h = queueCopy.front();
194 queueCopy.pop();
195
196 //resume the coroutine
197 h.resume();
198 } while (!queueCopy.empty());
199 }

367 virtual simtime_t updateTime(simtime_t TimePassed)
368 {
369 #ifdef DBG_CLOCK_H
370 DEBUG << "CLOCK_H(" << id() << ") updateTime(" << TimePassed << ")\n";
371 #endif
372
373 if(_running)
374 {
375 //Subtract time-passed from the time until the next event
376 _timeToNextEvent -= TimePassed;
377
378 //_TimeToNextEvent should never be negative
379 assert(_timeToNextEvent >= 0);
380
381 //It's possible that there are some small rouding errors in the
382 //calculation so therefore use a window of the given precision and 0
383 if (_timeToNextEvent >= 0 && _timeToNextEvent < _precision)
384 {
385 toggle();
386 }
387 }
388
389 return _timeToNextEvent;
390 }

406 void waitEdge(eClockEdge edge, coroutine_handle<> h)
407 {
408 switch (edge)
409 {
410 case eClockEdge::positive:
411 #ifdef DBG_CLOCK_H
412 DEBUG << "CLOCK_H(" << id() << ") wait positive edge\n";
413 #endif
414 posedgeQueue.push(h);
415 break;
416
417 case eClockEdge::negative:
418 #ifdef DBG_CLOCK_H
419 DEBUG << "CLOCK_H(" << id() << ") wait negative edge\n";
420 #endif
421 negedgeQueue.push(h);
422 break;
423
424 default:
425 FATAL << "Unkown clock edge \n";
426 break;
427 }
428 }
