188 T returnValue;

192 returnValue = *_defaultValue;
193 }
194 else if(!isSet() || (idx >= optionCount()))

196 std::string message;

199 message.append("option not set: ");

209 message.append("-" + getShortName());

211 else
212 {
213 message.append("--" + getLongName());

216 FATAL << message;
217 }
218 else
219 {
220 returnValue = _value[idx];

326 template <class T>
327 inline void cValueOption<T>::parse(cOptionName what_name, const char* value)
328 {
329 size_t parsed_value = 0;
330 std::string strValue;
331 if (value != nullptr)
332 {
333 strValue = value;
334 }
335
336 std::stringstream ss(strValue);
337 int valuesRead = 0;
338
339 while (ss.good())
340 {
341 if (ss.peek() != EOF)
342 {
343 ss >> parsed_value;
344 }
345 else
346 {
347 break;
348 }
349
350 valuesRead++;
351 }
352
353 if (ss.fail())
354 {
355 if(what_name == cOptionName::shortOption)
356 {
357 FATAL << "Invalid argument for: " << this->getShortName();
358 }
359 else
360 {
361 FATAL << "Invalid argument for: " << this->getLongName();
362 }
363 }
364
365 _value.push_back(parsed_value);
366 }
