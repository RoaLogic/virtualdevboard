147 virtual void push_back(T& data, bool OverwriteOnFull=true) {
148 std::lock_guard<std::mutex> lock(_mutex);
149
150 //check full and OverWriteOnFull
151 if (full() && !OverwriteOnFull) return;
152
153 //Insert element in buffer
154 _buffer[_head++] = data;
155
156 //check buffer overrung
157 if (_head == max_size()) _head = 0;
158
159 //update flags
160 _empty = false;
161 _full = _head == _tail;
162 }

169 virtual T pop_front() {
170 std::lock_guard<std::mutex> lock(_mutex);
171
172 //return empty element if the buffer is empty
173 if (empty()) return T();
174
175 //retrieve element from buffer
176 auto element = _buffer[_tail++];
177
178 //check buffer overrun
179 if (_tail == max_size()) _tail = 0;
180
181 //update flags
182 _full = false;
183 _empty = _tail == _head;
184
185 //return result
186 return element;
187 }
