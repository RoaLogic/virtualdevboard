140 virtual sCoRoutineHandler<bool> read(PADDR_t address, PDATA_t* buffer, unsigned burstCount=1)
141 {
142 bool result = false;
143 uint8_t addressOffset = sizeof(PDATA_t); // Determine the size of a single element so we can advance the address
144
145 if(!this->busy())
146 {
147 this->transactionStart();
148 this->_error = false;
149
150 // Perform the number of transaction passed by the user
151 for (size_t i = 0; i < burstCount; i++)
152 {
153 #ifdef DBG_BUSAPB4_H
154 DEBUG << "APB4 bus(" << this->id() << ") read from " << hex << unsigned(address + (addressOffset * i)) << '\n';
155 #endif
156
157 // Here we are at the IDLE phase
158 // Now we set the signals accordingly
159 PENABLE = L;
160 PWRITE = L;
161 PADDR = (address + (addressOffset * i)); // Set the address, add the offset onto the address
162 PSEL = H;
163
164 // Wait for the positive edge of the clock to transition into the access phase
165 waitPosEdge(PCLK);
166
167 // Set PENABLE high, so that we transition into the access phase with the next clock cycle
168 PENABLE = H;
169
170 waitPosEdge(PCLK);
171
172 // Wait until PREADY is in the right state and then transition out of the transaction
173 // PENABLE is already at the right state, since we set it at that point
174 // Since this is a single transaction it ends here.
175 while (PREADY == L) waitPosEdge(PCLK);
176
177 PENABLE = L;
178 buffer[i] = PRDATA;
179 }
180
181 // PSEL is only set when we transition back to the idle state, which is when there are no more transactions
182 PSEL = L;
183
184 this->_error = (PSLVERR == H);
185 result = !this->_error; // Set the result false in case we had a bus error
186
187 this->transactionEnd();
188 }
189 else
190 {
191 FATAL << "APB4 bus(" << this->id() << ") in busy state\n";
192 }
193
194 co_return result;
195 }
