101 virtual void tick(void) const
102 {
103 #ifdef DBG_TESTBENCH_H
104 DEBUG << "TESTBENCH_H - tick()" << std::endl;
105 #endif
106
107 //There should be at least 1 clock
108 assert (!_clkMgr->empty());
109
110 /*
111 * This is the correct order
112 * 1. eval design (this causes all @posedge to trigger)
113 * 2. trace->dump (previously evaluated values)
114 * 3. toggle clock and update time
115 * this resumes any waiting routines
116 * 4. eval design (this causes all @posedge to trigger)
117 */
118 //eval logic
119 _core->eval();
120
121 //dump trace
122 if (_traceActive && _trace)
123 {
124 simtime_t time = _clkMgr->getTime();
125 _trace->dump( (vluint64_t)(time.s() / _timeprecision) );
126 }
127
128 //tick() clocks and eval logic
129 _clkMgr->tick();
130
131 _core->eval();
132 }

141 }

200 cTestBench(VerilatedContext* context, bool traceActive) :
201 _context(context),
202 _finished(false),
203 _traceActive(traceActive)
204 {
205 if(traceActive)
