148 //streaming operator
149 inline std::ostream& operator<<(std::ostream& out, const simtime_t& t)
150 {
151 if (t > simtime_t::seconds_per_year ) { return out << t.year() << "years"; }
152 if (t > simtime_t::seconds_per_day ) { return out << t.day() << "days"; }
153 if (t > simtime_t::seconds_per_hour ) { return out << t.hour() << "hours"; }
154 if (t > simtime_t::seconds_per_minute) { return out << t.minute() << "minutes"; }
155 if (t > 1.0 ) { return out << t.s() << "sec"; }
156 if (t > 1.0E-3 ) { return out << t.ms() << "ms"; }
157 if (t > 1.0E-6 ) { return out << t.us() << "us"; }
158 if (t > 1.0E-9 ) { return out << t.ns() << "ns"; }
159 if (t > 1.0E-12 ) { return out << t.ps() << "ps"; }
160 if (t > 1.0E-15 ) { return out << t.fs() << "fs"; }
161
162 return out << t.as() << "as";
163 }
