182 int getRadix()
183 {
184 switch (lexer->getToken())

186 case tokBin:
187 return 2;
188

208 bool parse()
209 {
210 for (;;)
211 {
212 switch (state)
213 {
219 case (stateHeader):
220 switch (lexer->getToken())
221 {
222 case tokDepth:
223 getEquals();
224 depth = getNumber();
225 getSemicolon();
226 break;
227
228 case tokWidth:
229 getEquals();
230 width = getNumber();
231 getSemicolon();
232 break;
233
234 case tokAddressRadix:
235 getEquals();
236 addressRadix = getRadix();
237 getSemicolon();
238 break;
239
240 case tokDataRadix:
241 getEquals();
242 dataRadix = getRadix();
243 getSemicolon();
244 break;
245
246 case tokContent:
247 if (lexer->getToken() != tokBegin)
248 throw ParserException("BEGIN expected");
249 state = stateAddress;
250 break;
251
252 default:
253 throw ParserException("Unknown entry in MIF header section");
254 }
255 break;
256
257
263 case (stateAddress):
264 lexer->setRadix(addressRadix);
265 switch (lexer->getToken())
266 {
267 case tokNumber:
268 addressStart = lexer->getNumber();
269 addressRange = 0;
270 getColon();
271 state = stateData;
272 break;
273
274 case tokBracketLeft:
275 {
276 if (lexer->getToken() != tokNumber)
277 throw ParserException("Start of address range expected");
278
279 long addressLo = lexer->getNumber();
280
281 if (lexer->getToken() != tokDotDot)
282 throw ParserException("Address range expected '...'");
283
284 if (lexer->getToken() != tokNumber)
285 throw ParserException("End of address range expected");
286
287 long addressHi = lexer->getNumber();
288 if (addressHi < addressLo)
289 throw ParserException("Address high-range less than low-range");
290
291 addressStart = addressLo;
292 addressRange = addressHi - addressLo +1;
293 getBracketRight();
294 getSemicolon();
295 state = stateData;
296 }
297 break;
298
299 case tokEnd:
300 state = stateEof;
301 break;
302
303 default:
304 throw ParserException("Address expected");
305 }
306 break;
307
308
309 case (stateData):
310 lexer->setRadix(dataRadix);
311 switch (lexer->getToken())
312 {
313 case tokNumber:
314 //create new <address,data> pair
315 parseData.address = addressStart;
316 parseData.data = lexer->getNumber();
317 dataBuffer.push_back(parseData);
318 addressStart++;
319 addressRange--;
320 break;
321
322 case tokSemicolon:
323 state = stateAddress;
324 break;
325
326 default:
327 throw ParserException("Data expected");
328 }
329 break;
330
331
332 case (stateEof):
333 return true;
334 }
335 }
336 }
