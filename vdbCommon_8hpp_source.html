367 uint32_t iterator = 0;

369 for(auto ref : _referencePointers)

397 static void processVerilatorEvent(svScope scope, uint32_t event)
398 {
399 bool found = false;
400
401 // Loop all registered vdb components
402 for (const sVdbMap& ref : _referencePointers)
403 {
404 // Check if we found the component
405 if(ref.scope == scope)
406 {
407 // Call the callback with the event value
408 ref.reference->verilatorCallback(event);
409 found = true;
410 break;
411 }
412 }
413
414 if(!found)
415 {
416 WARNING << "VDB: Event on non registered module: " << svGetNameFromScope(scope) << " \n";
417 }
418 }
