248 static void processVerilatorEvent(svScope scope, uint32_t event)
249 {
250 bool found = false;
251
252 // Loop all registered vdb components
253 for (const sVdbMap& ref : _referencePointers)
254 {
255 // Check if we found the component
256 if(ref.scope == scope)
257 {
258 // Call the callback with the event value
259 ref.reference->verilatorCallback(event);
260 found = true;
261 break;
262 }
263 }
264
265 if(!found)
266 {
267 WARNING << "VDB: Event on non registered module: " << svGetNameFromScope(scope) << " \n";
268 }
269 }
